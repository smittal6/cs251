\SetKwProg{Fn}{}{\{}{\}}\SetKwFunction{myMerge}{void MergeSort}%
\Fn{\myMerge{X}}{ 
 \If{$Size(X)>1$}{
    MergeSort(X[0,...,\(\frac{n}{2}\)])\;
    MergeSort(X[\(\frac{n}{2}\)+1,...,n-1])\;
    Merge(X[0,..,\(\frac{n}{2}\)],X[\(\frac{n}{2}\)+1,...,n-1])\;
    }
 }

 
 \SetKwProg{Fn}{}{\{}{\}}\SetKwFunction{myRecurs}{int* Merge}%
 \Fn{\myRecurs{A,B}} {
    \ $i\leftarrow 0$ , $j\leftarrow 0$ , $t\leftarrow 0$ \;
    Initialise an array Y \;
    \While{$i+j<n$}{
        \eIf{$A[i]<=B[j]$}{
            $Y[t]\leftarrow A[i]$ \;
            $i\leftarrow i+1$ \;
            \If{$i=Size(A)$}{
                $A[i]\leftarrow infinity$ \;
            }
        }{
            $Y[t]\leftarrow B[j]$ \; 
            $j\leftarrow j+1$ \;
            \If{$j=Size(B)$}{
                $B[j]\leftarrow infinity$ \;
            }
        }
        $t\leftarrow t+1$ \;
        \KwRet{Y} \;
    }
 
 }

\end{algorithm}
